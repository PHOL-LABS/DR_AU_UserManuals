\chapter{Условия эксплуатации и требования безопасности}\label{ch:safety}

\section{Климатические условия}
\begin{itemize}
    \item Рабочая температура: от \(-40\,^{\circ}\mathrm{C}\) до \(+70\,^{\circ}\mathrm{C}\).
    \item Допустимая относительная влажность: до 95~\%.
    \item Разрешена круглогодичная эксплуатация внутри автомобиля.
\end{itemize}

\section{Электробезопасность}
\begin{itemize}
    \item Монтаж и обслуживание выполняются при отключённом питании.
    \item Не допускается эксплуатация с повреждённой проводкой или разъёмами.
\end{itemize}
